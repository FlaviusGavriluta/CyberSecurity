# Secret Directory Extraction

This document summarizes the steps taken to decrypt and extract the contents of the hidden `.secret` directory using the previously obtained FLAG `THM{w3lcome_2_A0c_2025}`.


# Navigate to the hidden secret directory
```
root@tbfc-web01:/home/eddi_knapp$ cd /home/eddi_knapp/.secret
```

# List contents
```
root@tbfc-web01:/home/eddi_knapp/.secret$ ls -la
total 420
drwxrwxr-x  2 eddi_knapp eddi_knapp   4096 Dec  1 08:32 .
drwxr-x--- 18 eddi_knapp eddi_knapp   4096 Dec  1 08:52 ..
-rw-------  1 eddi_knapp eddi_knapp 419589 Dec  1 08:32 dir.tar.gz.gpg
```

# Decrypt the GPG-encrypted archive using the FLAG as passphrase
```
root@tbfc-web01:/home/eddi_knapp/.secret$ gpg -d --batch --passphrase "THM{w3lcome_2_A0c_2025}" dir.tar.gz.gpg > dir.tar.gz
gpg: AES256.CFB encrypted data
gpg: encrypted with 1 passphrase
```

# Verify decrypted archive
```
root@tbfc-web01:/home/eddi_knapp/.secret$ ls -la
total 832
drwxrwxr-x  2 eddi_knapp eddi_knapp   4096 Dec  4 20:19 .
drwxr-x--- 18 eddi_knapp eddi_knapp   4096 Dec  1 08:52 ..
-rw-r--r--  1 root       root       419499 Dec  4 20:19 dir.tar.gz
-rw-------  1 eddi_knapp eddi_knapp 419589 Dec  1 08:32 dir.tar.gz.gpg
```

# Extract the tar.gz archive
```
root@tbfc-web01:/home/eddi_knapp/.secret$ tar -xzvf dir.tar.gz
dir/
dir/sq1.png
```

# Verify extracted contents
```

drwxrwxr-x  2 eddi_knapp eddi_knapp   4096 Dec  1 08:25 dir

```

root@tbfc-web01:/home/eddi_knapp/.secret/dir$ ls -la
total 420
drwxrwxr-x 2 eddi_knapp eddi_knapp   4096 Dec  1 08:25 .

```

```
